//Einstiegsmethode zur Steuereung des NPC
tree("BeAlive")
	fallback
		tree("Sit")
		tree("Stroll")

tree("TalkPC")
	sequence
		 UnsetDialogPartnerPC
		 IsPCInColliders
		 ISPCInTalkDist
		 StopMove
		 RotateToPC
		 SetDialogPartnerPC

tree("Sit")
	sequence
		IsSitVisible
		tree("ApproachSitOpportunity")
		StopMove
		RotateSit
		Wait(100)
		Sit
		Wait(300)
		StopSit
		Wait(300)
		Fail

tree("ApproachSitOpportunity")
	fallback
		repeat
			while not IsSitReached
				ApproachOpportunity
		Succeed


tree("Stroll")
    Stroll()
