//Einstiegsmethode zur Steuereung des NPC
tree("BeAlive")
	fallback
		tree("Talk")
		tree("Stroll")

tree ("Talk")
	sequence 
		IsTalkPartnerVisible
		tree("Approach")
		StopMove
		Random
			Wait(5)
			Wait(10)
			Wait(40)
			Wait(1)
		Talk
		Wait(100)
		RemovePartnerFromSight
		StopTalk
		Fail
	
tree("Approach")
	fallback
		repeat
			while not IsTalkPartnerReached
				ApproachTalkPartner
		Succeed

tree("Stroll")
    Stroll()
